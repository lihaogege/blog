npm -v				//查询版本号
npm install jquery			//按照jquery包
npm install jquery@x.y.z			//按照jquery包指定版本
npm uninstall  <包名称>	卸载包
npm list   npm ls 列出已安装模块
npm config ls   查看npm配置
npm config ls -l  查看npm所有配置
